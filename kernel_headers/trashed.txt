#Android makefile to build kernel as a part of Android Build

KERNEL_OUT := $(TARGET_OUT_INTERMEDIATES)/KERNEL_OBJ
KERNEL_HEADERS_INSTALL := $(KERNEL_OUT)/usr

PRODUCT_COPY_FILES += usr:KERNEL_OBJ/usr

$(KERNEL_OUT):
	echo "doing kernel_obj"
	mkdir $(TARGET_OUT_INTERMEDIATES)/KERNEL_OBJ

$(KERNEL_HEADERS_INSTALL): usr
	echo "doing usr"
	cp -a usr $(KERNEL_OUT)
